1. 判断一个正整数是不是质数？

```JavaScript
// function 1
function isPrimeOne(val) {
  // 质数需 >= 1
  if (val <= 1) {
    return false
  }
  // 2 和 3 是质数
  if ([2, 3].includes(val)) {
    return true
  }
  // 从 2 到 Math.sqrt(val) 依次判断是否为质数
  // 1：对于每个正整数而言，1 和自身肯定可以整除，所以从 2 开始
  // 2：只需要循环到平方根（非质数可以理解为 min * max，其中 min === max 时即该数可以被开方时就是 min 最大的时候，所以只需要循环到 Math.sqrt(val）
  // 示例：36 = 1 * 36 or 2 * 18 or 3 * 12 or 4 * 9 or 6 * 6，其中 min 最大为 6，max 最小也为 6
  for (let i = 2; i <= Math.sqrt(val); i++) {
    if (val % i === 0) {
      return false
    }
  }
  return true
}

// function 2
function isPrimeTwo(val) {
  // 质数需 >= 1
  if (val <= 1) {
    return false
  }
  if (val === 2) {
    return true
  }
  // 从 2 到 val-1 依次判断是否为质数（因为从 2 开始即默认 2 为质数开销相对要更大
  for (let i = 2; i < val; i++) {
    if (val % i === 0) {
      return false
    }
  }
  return true
}

// 返回小于正整数 n 的质数数量（link（埃及筛）：https://leetcode.cn/problems/count-primes/
var countPrimes = function (n) {
  // 长度为 n 的数组，初始化全部为 1
  // 1 - 质数；0 - 合数
  const isPrime = new Array(n).fill(1)
  let ans = 0
  for (let i = 2; i < n; ++i) {
  // or
  // for (let i = 2; i <= n; i++) {
    if (isPrime[i]) {
      ans += 1
      for (let j = i * i; j < n; j += i) {
        isPrime[j] = 0
      }
    }
  }
  return ans
}
```

2. 给定 m、n，找其最大公约数及最小公倍数？
```JavaScript
function findMaxAndMin(m, n) {
  // 最大公约数
  let max = m
  // 最小公倍数
  let min = n
  // 确保 m > n
  if (m < n) {
    ;[max, min] = [n, m]
  }

  while (max % min !== 0) {
    ;[max, min] = [min, max % min]
  }
  console.log({ max: (m * n) / min, min: min })
  // 最大公约数 * 最小公倍数 = m * n
  return { max: (m * n) / min, min: min }
}
findMaxAndMin(1, 100)
```
